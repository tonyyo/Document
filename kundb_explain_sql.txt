mysql5.7 Explain 官方语法：
{EXPLAIN | DESCRIBE | DESC}
    tbl_name [col_name | wild]  // mysql和vitess都不支持后两种

{EXPLAIN | DESCRIBE | DESC}
    [explain_type]
    {explainable_stmt | FOR CONNECTION connection_id}  // 后者类似与show processlist 外加 指明ID

explain_type: {
    EXTENDED                      // 为了获得warnings信息，8.0舍去
  | PARTITIONS			  // 8.0舍去
  | FORMAT = format_name          // 这个也是为了show warinings而存在的
}

format_name: {
    TRADITIONAL
  | JSON
}

explainable_stmt: {
    SELECT statement
  | DELETE statement
  | INSERT statement
  | REPLACE statement
  | UPDATE statement
}


mysql8.0 Explain 官方语法：

{EXPLAIN | DESCRIBE | DESC}
    tbl_name [col_name | wild]

{EXPLAIN | DESCRIBE | DESC}
    [explain_type]
    {explainable_stmt | FOR CONNECTION connection_id}

{EXPLAIN | DESCRIBE | DESC} ANALYZE select_statement    

explain_type: {
    FORMAT = format_name
}

format_name: {
    TRADITIONAL
  | JSON
  | TREE    // 8.0 新添
}

explainable_stmt: {   // vitess只支持crud
    SELECT statement
  | TABLE statement     // 8.0 新添
  | DELETE statement
  | INSERT statement
  | REPLACE statement
  | UPDATE statement
}
